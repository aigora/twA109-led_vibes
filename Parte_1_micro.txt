void setup() {
  Serial.begin(9600);
  int i;
  for(i=0;i<=6;i++){
    pinMode(i,OUTPUT);
  }
  for (i=0;i<=6;i++){
    digitalWrite(i,HIGH);
    delay(100);
  };
  for (i=0;i<=6;i++){
    digitalWrite(i,LOW);
    delay(100);
  };
}
